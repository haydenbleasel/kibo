import { useState } from 'react';
import {
  CodeBlock,
  CodeBlockHeader,
  CodeBlockCopyButton,
  CodeBlockTabsContent,
  CodeBlockTabsList,
  CodeBlockTabsTrigger
} from '@/components/ui/kibo-ui/code-block';

const code = [{
  name: 'MyComponent.js',
  code: \`function MyComponent(props) {
  return (
    <div>
      <h1>Hello, {props.name}!</h1>
      <p>This is an example React component.</p>
    </div>
  );
}\`,
}]

const Example = () => {
  const [value, setValue] = useState(code[0].name);
  const activeCommand = code.find((command) => command.name === value);

  return (
    <div className="p-4">
      <CodeBlock value={value} onChange={setValue}>
        <CodeBlockHeader>
          <CodeBlockTabsList>
            {code.map((item) => (
              <CodeBlockTabsTrigger key={item.name} value={item.name} label={item.name} />
            ))}
          </CodeBlockTabsList>
          <CodeBlockCopyButton
            value={activeCommand?.code}
            onCopy={() =>
              console.log(\`Copied "\${activeCommand?.code}" to clipboard\`)
            }
            onError={() =>
              console.error(\`Failed to copy "\${activeCommand?.code}" to clipboard\`)
            }
          />
        </CodeBlockHeader>
        {code.map((item) => (
          <CodeBlockTabsContent key={item.name} value={item.name}>
            {item.code}
          </CodeBlockTabsContent>
        ))}
      </CodeBlock>
    </div>
  );
};

export default Example;`} />

## Features

- Syntax highlighting
- Line numbers
- Copy to clipboard
- Filename and tabbed view
- Highlighted lines
- Language detection
- Theme support
- Customizable styles
